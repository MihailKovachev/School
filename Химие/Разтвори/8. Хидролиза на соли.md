	**а) слаба основа и силна киселина**
	
	$$\ce{NH4^+Cl^- + H^+OH^- <=>} \underset{\text{слаба основа}}{\ce{NH4OH}} + \underset{\text{силна киселина}}{\ce{HCl}}$$
	
	$$\underbrace{\ce{NH4^+ + Cl^-}} + \underset{\leftrightarrows \ce{H2O}}{\underbrace{\ce{H^+ + OH^-}}} \leftrightarrows \underset{\leftrightarrows \ce{NH3 + H2O}}{\ce{NH4OH}} + \underbrace{\ce{H^+ + Cl^-}}$$
	
	$$\ce{NH4^+ +} \underbrace{\ce{H^+ + OH^-}} \leftrightarrows \underset{\leftrightarrows \ce{NH3 + H2O}}{\ce{NH4OH}} + \ce{H^+}$$
	
	**б) силна основа и слаба киселина**
	
	$$\ce{Na2^+CO3^2- + 2H^+OH^- <=> \underset{\text{силна основа}}{2Na^+OH^-} + \underset{\text{слаба киселина}}{H2^+CO3^2-}}$$
	
	$$\underbrace{\ce{2Na^+ + CO3^2-}} + \underset{\ce{\leftrightarrows 2H2O}}{\underbrace{\ce{2H^+ + 2OH^-}}} \leftrightarrows \underbrace{\ce{2Na^+ + 2OH^-}} + \underset{\ce{\leftrightarrows CO2 + H2O}}{\ce{H2CO3}}$$
